## Настройка подключения к базе данных MySQL
## Используется для статистики, сбора отзывов,
## администрирования и внутриигровой библиотеки.

## Включить ли SQL? Раскомментируйте для активации
SQL_ENABLED

## Адрес сервера MySQL. Примеры:
## localhost, 200.135.5.43, www.mysqldb.com и т.д.
ADDRESS localhost

## Порт MySQL (по умолчанию 3306).
PORT 3306

## Имя базы данных для всех SQL-функций.
FEEDBACK_DATABASE tgstation_schema

## Префикс для названий таблиц (для старых БД требуется "erro_").
## Пример:
## FEEDBACK_TABLEPREFIX SS13_
## Примечание: не меняет имена таблиц в БД автоматически!
FEEDBACK_TABLEPREFIX

## Логин для доступа к базе данных.
FEEDBACK_LOGIN root

## Пароль для доступа к базе данных.
FEEDBACK_PASSWORD root

## Таймаут асинхронных запросов (в секундах).
## 0 = бесконечное ожидание.
ASYNC_QUERY_TIMEOUT 10

## Таймаут блокирующих запросов (в секундах).
## Должен быть <= ASYNC_QUERY_TIMEOUT.
## 0 = бесконечное ожидание.
BLOCKING_QUERY_TIMEOUT 5

## Минимальное количество соединений в пуле.
## Увеличьте для серверов с высокой нагрузкой.
POOLING_MIN_SQL_CONNECTIONS 1

## Максимальное количество соединений с БД.
POOLING_MAX_SQL_CONNECTIONS 25

## Лимит одновременных асинхронных запросов.
## При превышении запросы ставятся в очередь.
MAX_CONCURRENT_QUERIES 25
